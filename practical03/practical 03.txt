01. TRAFFIC LIGHT SYSTEM..

void setup(){
  pinMode(9, OUTPUT);
  pinMode(8, OUTPUT);
  pinMode(7, OUTPUT);
}

void loop(){
  digitalWrite(7, HIGH);
  delay(5000);
  digitalWrite(7, LOW);

  digitalWrite(8, HIGH);
  delay(1000);
  digitalWrite(8, LOW);

  digitalWrite(9, HIGH);
  delay(2000);
  digitalWrite(9, LOW);

  digitalWrite(8, HIGH);
  delay(1000);
  digitalWrite(8, LOW);

}
